void main() {
	char a, b;
	a = 'f';
	b = 'g';

    if (a > b) {
        write 'a';
    }
    if (a >= b) {
        write 'b';
    }
    if (a < b) {
        write 'c';
    }
    if (a <= b) {
        write 'd';
    }
    if (a == b) {
        write 'e';
    }
    if (a != b) {
        write 'f';
    }
}
