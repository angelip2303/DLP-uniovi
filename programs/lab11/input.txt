char c;
int a;
double r;

void main() {
	int a, i;
	
	c='a';
	write c,'\n';

	a=1;
	i=(int)'0'*3;
	write a, ' ', i, '\n';
		
	read r;
	write (double)2*r, '\n';
	
	write (double)3-4.5, ' ', 
            (double)'0'/48.0, ' ',
            '0'%'1', ' ',
            '0'+'1', '\n';
	write 1<2, ' ',
	      4>=3, ' ',
	      5==5, ' ',
	      1&&2, ' ',
	      0||3, ' ',
	      !0, '\n';
}
